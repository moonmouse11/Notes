# Shell Scripts
***
## Basic shell commands
**`!#`** - шебанг. Первая строка для bash скрипта.
``` bash
# - коммантарий в bash.
echo # - выводит переменную в командную строкуs

# Переменные окружения
$BASH # полный путь до оболчки BASH
$BASH_VERSION # версия BASH
$HOME # домашний каталог
$HOSTNAME # имя компьютера
$RANDOM # случайное число
$OSTYPE # тип операционной системы
$PWD # текущий каталог
$PS1 # сррока приглашения
$UID # идентификатор пользователя
$USER # имя пользователя

# Создание персональное переменной окружения
export PATH=<значения>

# Вывод переданных аргументов в сценарии с помощью цикла
for n in $@
do 
	echo "$n"
done

# if-elif
if [условие] then
	[команда]
elif [условие] then
	[команда]
else
	[команда]
fi

# Операции сравнения
-lt # Меньше
-gt # Больше
-le # Меньше или равно
-ge # Больше или равно
-eq # Равно

# Условия для проверки файлов
-e file # истинно, если файл существует
-d directory # истинно, если директория существует
-x file # истинно, если файл исполняемый

exit 0 # Завершение исполнения скрипта
read [var] # чтение значение из командной строки.

# case_esac
case "$var" in
	case1) first_instruction ;;
	case2) second_instruction ;;
	case3) third_instruction ;;
	*) default_instruction ;;
esac

# Цикл for
for $var in $LIST;
do
	[тело_цикла]
done

# Цикл while
while [условие]
do
	[тело_цикла]
done

# Объявление пользовательской функции
[function_name] () 
{
	[тело_функции]
}
```
